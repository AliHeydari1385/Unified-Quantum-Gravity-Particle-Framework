UQCMF Data Preprocessing Module v1.0
------------------------------------
Downloads and prepares real solar X-ray and geomagnetic data for UQCMF analysis.

- Solar Data: Fetches SOHO/CDS (Coronal Diagnostic Spectrometer) data via SunPy.
- Auroral Proxy: Fetches Kp-index data from NOAA as a proxy for geomagnetic storm intensity.
"""

import sunpy.net.Fido as Fido
import sunpy.net.attrs as a
from sunpy.timeseries import TimeSeries
import pandas as pd
from datetime import datetime

class DataHandler:
    def __init__(self, start_time, end_time):
        self.start_time = start_time
        self.end_time = end_time
        print(f"Initializing data handler for period: {start_time} to {end_time}")

    def get_solar_xray_data(self):
        """
        Fetches GOES X-ray flux data, which is a good proxy for coronal activity.
        This is more readily available than specific spectral line data.
        """
        print("Querying GOES X-ray flux data...")
        # Search for GOES XRS (X-ray Sensor) data
        result = Fido.search(a.Time(self.start_time, self.end_time), a.Instrument.xrs, a.Satellite.goes)
        if not result:
            print("No GOES data found for the specified period.")
            return None
        
        # Download the data
        files = Fido.fetch(result)
        goes_ts = TimeSeries(files)
        
        # Resample to a consistent cadence (e.g., 1 minute) and select the long-wavelength channel
        df = goes_ts.to_dataframe()
        df_resampled = df['xrsb'].resample('1T').mean().dropna()
        
        print(f"âœ… Successfully fetched and processed {len(df_resampled)} solar X-ray data points.")
        return df_resampled

    def get_geomagnetic_data(self):
        """
        Fetches the Kp-index, a global measure of geomagnetic activity, which correlates with auroral intensity.
        We can use a library like 'geomagpy' or fetch directly from NOAA.
        """
        print("Fetching Kp-index data from NOAA...")
        # For simplicity, we'll construct a URL to a standard text file from NOAA's FTP server.
        # This needs to be adapted based on the year/month.
        url = f"https://services.swpc.noaa.gov/text/daily-geomagnetic-indices.txt"
        try:
            # Read data, skipping header comments
            df = pd.read_csv(url, comment='#', header=None, delim_whitespace=True,
                             names=['Date', 'Time', 'Station_Count', 'Kp_Sum', 'Ap_Sum', 'Kp', 'Ap', 'C9'])
            
            # Combine Date and Time into a proper datetime index
            df['datetime'] = pd.to_datetime(df['Date'] + ' ' + df['Time'])
            df.set_index('datetime', inplace=True)
            
            # Filter for the desired time range
            df_filtered = df['Kp'][(df.index >= self.start_time) & (df.index <= self.end_time)]
            
            print(f"âœ… Successfully fetched and processed {len(df_filtered)} Kp-index data points.")
            return df_filtered
        except Exception as e:
            print(f"Error fetching geomagnetic data: {e}")
            return None
